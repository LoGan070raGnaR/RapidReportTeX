	\newcommand{\myborder}{%
		\begin{tikzpicture}[remember picture,overlay]
			% Draw a rectangle for the outer border
			\draw[line width=1pt]
			($(current page.north west) + (0.45in,-0.45in)$) rectangle
			($(current page.south east) + (-0.5in,0.45in)$);

			($(current page.north west) + (0.85in,-0.85in)$) -- ($(current page.north east) + (-0.9in,-0.85in)$)

		\end{tikzpicture}%
	}
